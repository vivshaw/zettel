	- Refinements can be conditional!
	  ```Ruby
	  module StringManipulator
	    refine String do
	      def reverse_upcase
	        reverse.upcase
	      end
	    end
	  end
	  
	  class Example
	    using StringManipulator if RUBY_VERSION >= "2.6.0"
	  
	    def manipulate_string(input)
	      input.reverse_upcase
	    end
	  end
	  ```
